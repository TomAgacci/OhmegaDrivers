cmake_minimum_required(VERSION 3.20)
project(GammaWCS LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

add_executable(GammaWCS
    main.cpp
)

target_link_options(GammaWCS PRIVATE "/SUBSYSTEM:WINDOWS")

target_link_libraries(GammaWCS PRIVATE
    user32
    gdi32
    mscms
)
